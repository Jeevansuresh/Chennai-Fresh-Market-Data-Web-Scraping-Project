-- Create database
CREATE DATABASE IF NOT EXISTS vegetables;

-- Use the created database
USE vegetables;

-- Drop the table if it already exists
DROP TABLE IF EXISTS VEG_DATA;

-- Create the VEG_DATA table
CREATE TABLE VEG_DATA (
    date DATE,
    vegetable VARCHAR(100),
    wholesale_price DECIMAL(10, 2),
    retail_min DECIMAL(10, 2),
    retail_max DECIMAL(10, 2),
    mall_min DECIMAL(10, 2),
    mall_max DECIMAL(10, 2),
    units VARCHAR(10)
);

-- Optional: View inserted data
SELECT * FROM VEG_DATA;

-- Optional: Clear table data
DELETE FROM VEG_DATA;
